# Real-time vitals tracker
class VitalsMonitor:
  def __init__(self):
    self.temp = 98.6
    self.spo2 = 98
    self.bpm  = 72
 
  def read(self):
    return {
      "temp": self.temp,
      "spo2": self.spo2,
      "bpm":  self.bpm
    }
